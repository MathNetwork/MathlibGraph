\section{The Structure of Mathlib}

\subsection{Module Names}

A \emph{module name} in Mathlib is a dot-separated sequence of identifiers, such as
\[
  \texttt{Mathlib.Algebra.Group.Defs}
\]
We write $\mathcal{M}$ for the set of all module names in Mathlib. As of commit \texttt{534cf0b} (2 February 2026), $|\mathcal{M}| = 7{,}563$.

The \emph{depth} of a module name is the number of components. For example, $\texttt{Mathlib.Algebra.Group.Defs}$ has depth~$4$. The maximum depth in Mathlib is~$12$.

\subsection{The Prefix Tree}

Module names are organized hierarchically. We say $m$ is a \emph{prefix} of $m'$ if $m'$ extends $m$ by additional components. For example, $\texttt{Mathlib.Algebra}$ is a prefix of $\texttt{Mathlib.Algebra.Group.Defs}$.

The prefix relation induces a rooted tree $T = (\mathcal{M}, E_T)$ where:
\begin{itemize}
  \item The root is $\texttt{Mathlib}$
  \item There is an edge $m \to m'$ if $m'$ extends $m$ by exactly one component
\end{itemize}

The top-level directories (children of the root) include \texttt{Algebra} (1257 files), \texttt{CategoryTheory} (982 files), \texttt{Analysis} (755 files), etc.

\subsection{Graph-Theoretic Definitions}

\begin{definition}[Directed graph]
A \emph{directed graph} is a pair $G = (V, E)$ where $V$ is a set (called the \emph{vertex set}) and $E \subseteq V \times V$ (called the \emph{edge set}).
\end{definition}

\begin{definition}[Path]
A \emph{path} in a directed graph $G = (V, E)$ is a sequence of vertices $(v_0, v_1, \ldots, v_k)$ such that $(v_i, v_{i+1}) \in E$ for all $0 \leq i < k$. The \emph{length} of the path is $k$.
\end{definition}

\begin{definition}[Cycle]
A \emph{cycle} is a path $(v_0, v_1, \ldots, v_k)$ with $k \geq 1$ and $v_k = v_0$.
\end{definition}

\begin{definition}[Directed acyclic graph]
A directed graph $G = (V, E)$ is a \emph{directed acyclic graph (DAG)} if $G$ contains no cycle.
\end{definition}

\subsection{The Import Graph}

Let $\mathcal{M}$ be the set of Mathlib module names. For each $m \in \mathcal{M}$, let $\mathrm{imports}(m) \subseteq \mathcal{M}$ denote the set of modules directly imported by $m$.

\begin{definition}[Import graph]
The \emph{import graph} of Mathlib is $G_{\mathrm{import}} = (\mathcal{M}, E)$ where
\[
  E = \{(m_1, m_2) \in \mathcal{M} \times \mathcal{M} \mid m_2 \in \mathrm{imports}(m_1)\}.
\]
\end{definition}

\begin{proposition}
$G_{\mathrm{import}}$ is a directed graph.
\end{proposition}

\begin{proof}
$\mathcal{M}$ is a finite set (of $7{,}563$ module names), and $E \subseteq \mathcal{M} \times \mathcal{M}$ by construction.
\end{proof}

\begin{proposition}
$G_{\mathrm{import}}$ is acyclic.
\end{proposition}

\begin{proof}
The Lean compiler processes modules in a well-defined compilation order: before compiling a module $m$, every module in $\mathrm{imports}(m)$ must already be compiled. Since Mathlib compiles successfully, there exists an injective function $\sigma \colon \mathcal{M} \to \mathbb{N}$ (the compilation order) such that
\[
  m_2 \in \mathrm{imports}(m_1) \implies \sigma(m_2) < \sigma(m_1).
\]
Suppose for contradiction that $G_{\mathrm{import}}$ contains a cycle $(m_0, m_1, \ldots, m_k = m_0)$ with $k \geq 1$. Then $(m_i, m_{i+1}) \in E$ for each $i$, so $\sigma(m_0) > \sigma(m_1) > \cdots > \sigma(m_k) = \sigma(m_0)$, a contradiction.
\end{proof}

\begin{corollary}
$G_{\mathrm{import}}$ is a DAG.
\end{corollary}
