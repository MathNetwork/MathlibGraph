\section{Evolution Metrics}

Let $G^{(t)} = (V^{(t)}, E^{(t)})$ be the graph snapshot at time~$t$.

\subsection{Core growth rate}

\begin{definition}[Core growth rate]
Define the core as the set of high in-degree nodes:
$\mathrm{Core}^{(t)} = \{v \in V^{(t)} \mid \deg^-(v) > \theta\}$.
\[
  \gamma_{\mathrm{core}}(t) = \frac{|\mathrm{Core}^{(t+1)}| - |\mathrm{Core}^{(t)}|}{|\mathrm{Core}^{(t)}|}.
\]
\end{definition}

\subsection{Dependency distance entropy}

\begin{definition}[Dependency distance entropy]
Let $P_t(d)$ be the distribution of edge lengths~$d$ (topological level difference) at time~$t$.
\[
  H_{\mathrm{dep}}(t) = -\sum_d P_t(d) \log P_t(d).
\]
\end{definition}
