\section{Global Geometric Quantities}

\subsection{Topological level}

\begin{definition}[Topological level]
Define the level function $\ell \colon V \to \mathbb{N}$:
\[
  \ell(v) =
  \begin{cases}
    0 & \text{if } \deg^-(v) = 0, \\
    1 + \max_{u \in N^-(v)} \ell(u) & \text{otherwise}.
  \end{cases}
\]
\end{definition}

\subsection{Build depth}

\begin{definition}[Build depth]
$\mathrm{Depth}(G) = \max_{v \in V} \ell(v)$.
\end{definition}

\subsection{Edge density}

\begin{definition}[Edge density]
\[
  \rho(G) = \frac{|E|}{|V|(|V|-1)}.
\]
\end{definition}
