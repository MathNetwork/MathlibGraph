\section{Basic Structure}

Let $G = (V, E)$ be one of the above models.

\subsection{Adjacency}

\begin{definition}[Out-neighbourhood]
$N^+(v) = \{u \in V \mid (v, u) \in E\}$
\end{definition}

\begin{definition}[In-neighbourhood]
$N^-(v) = \{u \in V \mid (u, v) \in E\}$
\end{definition}

\begin{definition}[Out-degree / In-degree]
$\deg^+(v) = |N^+(v)|$, \quad $\deg^-(v) = |N^-(v)|$.
\end{definition}

\subsection{Reachability}

Let $v \leadsto u$ denote the existence of a directed path from $v$ to $u$.

\begin{definition}[Downstream set]
$\mathrm{Down}(v) = \{u \in V \mid v \leadsto u\}$
\end{definition}

\begin{definition}[Upstream set]
$\mathrm{Up}(v) = \{u \in V \mid u \leadsto v\}$
\end{definition}
