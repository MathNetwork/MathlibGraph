\section{Modularity and Community Structure}

Let $\mathcal{C} = \{C_1, \ldots, C_k\}$ be a partition of~$V$.

\subsection{Modularity}

\begin{definition}[Modularity]
For the undirected projection $G^{\mathrm{und}}$ (ignoring edge directions):
\[
  Q(\mathcal{C}) = \frac{1}{2|E|} \sum_{i} \sum_{u,v \in C_i}
    \left( A_{uv} - \frac{k_u k_v}{2|E|} \right),
\]
where $A$ is the adjacency matrix and $k_v = \deg(v)$.
\end{definition}

\subsection{Boundary crossing ratio}

\begin{definition}[Boundary crossing ratio]
Let $\pi \colon V \to \mathcal{D}$ be the directory-structure mapping
(e.g.\ \texttt{Topology}, \texttt{Algebra}, etc.). Then
\[
  \mathrm{Cross}(G) = \frac{|\{(u,v) \in E \mid \pi(u) \neq \pi(v)\}|}{|E|}.
\]
\end{definition}
